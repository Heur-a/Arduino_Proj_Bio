34template< typename T >
35T* alReves(T* p, int n) {
36 T aux;
37
38 for (int i = 0; i < n / 2; i++) {
39 aux = p[i];
40 p[i] = p[n - i - 1];
41 p[n - i - 1] = aux;
42 }
43 return p;
44} // ()
